function createIncrement() {
    let count=0;
    function increment() {count++;}
        let message=`Count is ${count}`;
        console.log(message);
        function log()
    {
    console.log(message);                                                                      }
    return[increment,log]
};
const[increment,log] =createIncrement();
increment();
increment();
increment();
log();
